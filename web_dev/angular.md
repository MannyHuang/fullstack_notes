# angular 

## cli
```cli
# install cli
npm install -g @angular/cli

# create new app
ng new [appName]

# serve the app
ng serve -o

# genereate component
ng genereate component [compName]

# generate component shorthand
ng g c [compName] --spec false

# generate pipe in specified module
ng g pipe [pipeName] -m [moduleName]

# add pwa functionality
ng add @angular/pwa

# build the app
ng build --prod

# search doc
ng doc [searchTerm]
```
